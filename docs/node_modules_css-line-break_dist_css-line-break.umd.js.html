                classType -= LETTER_NUMBER_MODIFIER;

                    return types.push(AL);

                    return types.push(prev);

                indices.push(index);
                return types.push(AL);

            if (classType === CJ) {
                return types.push(lineBreak === 'strict' ? NS : ID);
            }
            if (classType === SA) {
                return types.push(AL);
            }
            if (classType === AI) {
                return types.push(AL);

                }

                }
            }
